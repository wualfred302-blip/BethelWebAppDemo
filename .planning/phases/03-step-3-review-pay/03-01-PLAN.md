    2. **PRICING_TABLE constant** — array of 27 PricingRow objects from the data above. For "0 - 20": minSqm=0, maxSqm=20. For "901 - 1000": minSqm=901, maxSqm=1000.

    3. **NATURE_TO_CLASS mapping** — Record<string, 1 | 2> mapping each natureOfBusiness option to its class. Use the Class I/II lists from context.

    4. **lookupPremium function:**
    ```typescript
    export function lookupPremium(floorAreaSqM: string, natureOfBusiness: string) {
      const sqm = parseInt(floorAreaSqM, 10) || 0;
      const row = PRICING_TABLE.find(r => sqm >= r.minSqm && sqm <= r.maxSqm);
      if (!row) return null;
      const cls = NATURE_TO_CLASS[natureOfBusiness] ?? 2; // default Class II
      return {
        limitOfLiability: row.limitOfLiability,
        netPremium: cls === 1 ? row.classINet : row.classIINet,
        grossPremium: cls === 1 ? row.classIGross : row.classIIGross,
        class: cls,
      };
    }
    ```

    5. **formatPHP function** — formats a number as Philippine Peso:
    ```typescript
    export function formatPHP(amount: number): string {
      return new Intl.NumberFormat('en-PH', {
        style: 'currency',
        currency: 'PHP',
      }).format(amount);
    }
    ```

    Export all public functions and constants.
  </action>

      const now = new Date();
      const date = `${now.getFullYear()}${String(now.getMonth() + 1).padStart(2, '0')}${String(now.getDate()).padStart(2, '0')}`;
      const suffix = String(Math.floor(1000 + Math.random() * 9000));
      return `CGL-${date}-${suffix}`;
    }

    export function generateExpiryTime(hours: number = 6): string {
      const expiry = new Date(Date.now() + hours * 60 * 60 * 1000);
      return expiry.toLocaleTimeString('en-PH', { hour: 'numeric', minute: '2-digit', hour12: true });
    }
